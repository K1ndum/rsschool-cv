function createDreamTeam(arr) {
  if (!arr) {
    return false;
  }
  let array = [];
  for (let i = 0; i < arr.length; i++) {
    if (typeof(arr[i]) == "string") {
      array.push((arr[i].match(/\w/))[0].toLocaleUpperCase());  
    }
  }
  return array.sort().join('');
}
