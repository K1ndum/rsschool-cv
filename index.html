function revrot(str, sz) {
  if (sz === 0 || str === '') {
    return ''
  }
  let regexp = new RegExp('\\d{' +sz+ '}', 'g');
  let arr = str.match(regexp)
  if (arr === null) {
    return ''
  }
  let arrayP;
  let arrEnd = [];
  for (let i = 0; i < arr.length; i++) {
    let sum = 0;
    arrayP = arr[i].split('');
    let array = arr[i].split('');
    array = array.map((num) => Math.pow(num, 3));
    sum = array.reduce((sum, num) => sum+=num);
    if (sum % 2 == 0) {
      arrEnd.push(arrayP.reverse().join(''))
    } else {
      arrayP.push(arrayP[0]);
      arrayP.shift();
      arrEnd.push(arrayP.join(''));
    }
  }
  return (arrEnd.join(''));
}
